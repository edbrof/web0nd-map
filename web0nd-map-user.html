  <script src="https://unpkg.com/leaflet@1.9.3/dist/leaflet.js"></script>

  <script src="https://unpkg.com/leaflet-control-geocoder/dist/Control.Geocoder.js"></script>

  <script>

    const savedPins = JSON.parse(localStorage.getItem('userPins') || '[]');


    const map = L.map('map').setView([20, 0], 2);

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

      attribution: '&copy; OpenStreetMap contributors'

    }).addTo(map);


    const geocoder = L.Control.geocoder({ defaultMarkGeocode: false })

      .on('markgeocode', function(e) {

        dropPin(e.geocode.center, e.geocode.name);

        map.setView(e.geocode.center, 6);

      }).addTo(map);


    savedPins.forEach(({ lat, lng, name }) => dropPin([lat, lng], name));


    function dropPin(coords, label) {

      const marker = L.marker(coords).addTo(map).bindPopup(

        `<b>${label}</b><br><button onclick="removePin(${coords[0]}, ${coords[1]})">Remove</button>`

      );


      savedPins.push({ lat: coords[0], lng: coords[1], name: label });

      localStorage.setItem('userPins', JSON.stringify(savedPins));

    }


    function removePin(lat, lng) {

      const index = savedPins.findIndex(pin => pin.lat === lat && pin.lng === lng);

      if (index !== -1) {

        savedPins.splice(index, 1);

        localStorage.setItem('userPins', JSON.stringify(savedPins));

        location.reload();

      }

    }


    document.getElementById('submitBtn').addEventListener('click', async () => {

      const input = document.getElementById('cityInput').value;

      const cities = input.split(',').map(c => c.trim()).filter(c => c);

      for (const city of cities) {

        await fetch(`https://nominatim.openstreetmap.org/search?format=json&q=${encodeURIComponent(city)}`)

          .then(res => res.json())

          .then(data => {

            if (data[0]) {

              dropPin([+data[0].lat, +data[0].lon], city);

              map.setView([+data[0].lat, +data[0].lon], 6);

            }

          });

      }

      document.getElementById('intro').style.display = 'none';

    });

  </script>
